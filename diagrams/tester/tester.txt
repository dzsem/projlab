procedure Tester.Assert(assertfile: File, outputfile: File) is
    let err be a list of Strings
    String assert, String output;
    boolean found;

    for each non-comment line in assertfile
        found = false;
        assert = nextLine;
        for each non-comment line in outputfile
            output = nextLine;
            if assert == output
                found = true
                return
        if found == false
            add assert to err
    
    if err is empty
        print SUCCESS
    else
        print FAIL
        print err
